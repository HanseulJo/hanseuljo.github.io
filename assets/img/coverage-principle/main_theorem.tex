\noindent\textbf{Theorem 6.1} (Informal, $k\ge 2$)\textbf{.}~
\textit{Consider a} \textsc{2-Hop} \textit{task with a token set of size~$n$. For a uniformly randomly sampled train dataset $D$ of size $N$, consider a learner that generalizes within the $k$-coverage of $D$. Then, for any $\delta > 0$ and large enough $n$, the learner can achieve perfect ID generalization with probability at least $1-\delta$ if}
\vspace{-10pt}
\begin{align*}
N \gtrsim \max\left\{ \textcolor{RoyalBlue3}{n^{2.5 - \frac{1}{2k}}} (k! \ln n)^{\!\frac{1}{2k}}, \ln\frac1\delta \right\}. \tag{Upper Bound}
\end{align*}
\vspace{-10pt}

\noindent\textit{In contrast, with probability at least $1-\delta$, the learner \underline{cannot} achieve perfect ID generalization for some} \textsc{2-Hop} \textit{task if}
\vspace{-10pt}
\begin{align*}
n^2 \max \left\{\ln n, \ln\frac1\delta\right\} \lesssim N \lesssim \textcolor{Firebrick3}{n^{2.5 - \frac{1}{2k}}} ((k-1)! \ln n)^{\!\frac{1}{2k}}. \tag{Lower Bound}
\end{align*}
